  272.     /**
  273.      * CLICommander constructor.
  274.      * 
  275.      * Returns an instanced CLICommander object
  276.      */
  277.     public function CLICommander() {
  278.         // Check to see if we are using windows
  279.         if (substr(php_uname('s'),0,7) == 'Windows') $this->usingWindows = true;
  280.         
  281.         // Open our sockets
  282.         $this->outputSocket = @fopen($this->outputStream, 'w');
  283.         $this->inputSocket = @fopen($this->inputStream, 'r');
  284.         $this->errorSocket = @fopen($this->errorStream, 'w');
  285.         
  286.         if (!$this->usingWindows) {
  287.             // Check for xterm support
  288.             if (strpos($_SERVER['TERM'],'xterm') !== false) $this->xtermSupport = true;
  289.             
  290.             // Check for bash
  291.             $test = `which bash`;
  292.             if (!empty($test)) $this->bashSupport = true;
  293.         }
  294.         
  295.         // Assign the Argument Variables
  296.         $this->argc = $_SERVER['argc'];
  297.         $this->argv = $_SERVER['argv'];
  298.         
  299.         // Process any passed arguments
  300.         $this->ProcessArguments();
  301.     }

  361.     /**
  362.      * Get an array of the arguments passed and their associated values (if there are any)
  363.      * 
  364.      * Any arguments passed to the script without a preceeding "-" or "--" will show up in
  365.      * this array with numbered indexes.  All arguments passed with a preceeding "-" or "--"
  366.      * will show up with named indexes
  367.      * 
  368.      * <code>
  369.      * ./myScript.php -vad --file myFile.php
  370.      * <?php
  371.      * print_r($cli->GetArguments());
  372.      * ?>
  373.      * OUTPUT:
  374.      * array(
  375.      *     'v'    =>    1,
  376.      *     'a'    =>    1,
  377.      *     'd'    =>    1,
  378.      *     'file'    =>    'myFile.php',
  379.      *     0    =>    'myFile.php'
  380.      * )
  381.      * </code>
  382.      * 
  383.      * @return array 
  384.      */
  385.     public function GetArguments() {
  386.         $args = array();
  387.         foreach ($this->argumentsPassed as $i => $v) {
  388.             $args[$i] = (isset($this->argumentValues[$i])) ? $this->argumentValues[$i] : $v;
  389.         }
  390.         
  391.         return $args;
  392.     }

  396.      *

  566.     /**
  567.      * Write a string to the terminal with defined formatting
  568.      * 
  569.      * Please note that Write() does not append newline characters to the text entered.
  570.      * If you need a newline character, use WriteLine()
  571.      * 
  572.      * @param string $text The text to display
  573.      * @param string|integer|array$fgColor The color to use for the text displayed or a style array to use for all the formatting
  574.      * @param string|integer$bgColor The color to use for the background
  575.      * @param string $style The style to use for the text displayed
  576.      */
  577.     public function Write($text, $fgColor = null, $bgColor = null, $style = null) {
  578.         if (!$this->usingWindows) {
  579.             // Check our colors and styles
  580.             if (is_array($fgColor)) {
  581.                 // User passed a style array
  582.                 $style = (isset($fgColor['style']) && !empty($fgColor['style'])) ? $fgColor['style'] : $this->defaults['style'];
  583.                 $bgColor = (isset($fgColor['background']) && !empty($fgColor['background'])) ? $fgColor['background'] : $this->defaults['background'];
  584.                 $fgColor = (isset($fgColor['foreground']) && !empty($fgColor['foreground'])) ? $fgColor['foreground'] : $this->defaults['foreground'];
  585.             } else {
  586.                 // Check for individual options
  587.                 if ($fgColor == null) $fgColor = $this->defaults['foreground'];
  588.                 if ($bgColor == null) $bgColor = $this->defaults['background'];
  589.                 if ($style == null) $style = $this->defaults['style'];
  590.             }
  591.             
  592.             $format = $this->GetFormatString($fgColor, $bgColor, $style);
  593.             
  594.             // Save text with formatting escape sequence
  595.             $text = $format.$text;
  596.         }
  597.         
  598.         // Write our text to the output socket
  599.         $this->SystemWrite($text);
  600.         
  601.         if ($this->autoReset) $this->Reset();
  602.     }

  642.     /**
  643.      * Convert an RGB hex color to the closest xterm equivilent
  644.      * 
  645.      * @param string $rgbString The hex string to convert
  646.      * @param boolean $foreground Whether this is for a foreground or background color
  647.      * 
  648.      * @return integer 
  649.      */
  650.     private function ClosestXtermColor($rgbString, $foreground = true) {
  651.         // Replace # sign if there
  652.         $rgbString = str_replace('#','',strtoupper($rgbString));
  653.         
  654.         // Check the length
  655.         if (strlen($rgbString) != 6) {
  656.             if ($foreground) return 231; // White
  657.             return 16; // Black
  658.         }
  659.         
  660.         // Breakout the RGB colors
  661.         $r = hexdec(substr($rgbString,0,2));
  662.         $g = hexdec(substr($rgbString,2,2));
  663.         $b = hexdec(substr($rgbString,4,2));
  664.         
  665.         // Check for Greyscale color
  666.         if ($r == $g && $g == $b) {
  667.             $g = $this->ClosestXtermGrey($r);
  668.             $color = $g.$g.$g;
  669.         } else {
  670.             // Color
  671.             $color = $this->ClosestXtermOctet($r).$this->ClosestXtermOctet($g).$this->ClosestXtermOctet($b);
  672.         }
  673.         
  674.         return $this->xtermColors[$color];
  675.     }
  676.     
  677.     /**
  678.      * Convert a hex octet to the closest grey octet used for xterm colors
  679.      * 
  680.      * @param integer $g The integer value of the hex to be converted
  681.      * 
  682.      * @return string 
  683.      */
  684.     private function ClosestXtermGrey($g = 0) {
  685.         if ($g < 4) return '00';
  686.         if ($g > 243) return 'FF';
  687.  
  688.         $m = $g % 10;
  689.  
  690.         if ($m != 8) {
  691.             if ($m > 3 && $m < 8) {
  692.                 $g = $g + (8 - $m);
  693.             } else {
  694.                 switch ($m) {
  695.                     case 3:
  696.                         $g--;
  697.                     case 2:
  698.                         $g--;
  699.                     case 1:
  700.                         $g--;
  701.                     case 0:
  702.                         $g--;
  703.                     case 9:
  704.                         $g--;
  705.                 }
  706.             }
  707.         }
  708.         
  709.         unset($m);
  710.         $h = dechex($g);
  711.         unset($g);
  712.         
  713.         if (strlen($h) == 1) {
  714.             return '0'.$h;
  715.         } else {
  716.             return $h;
  717.         }
  718.     }
  719.     
  720.     /**
  721.      * Get the closest xterm color octet for the octet supplied
  722.      * 
  723.      * @param integer $c 
  724.      * 
  725.      * @return string 
  726.      */
  727.     private function ClosestXtermOctet($c = 0) {
  728.         if ($c >= 0 && $c < 47) {
  729.             return '00';
  730.         } elseif ($c > 46 && $c < 116) {
  731.             return '5F';
  732.         } elseif ($c > 115 && $c < 156) {
  733.             return '87';
  734.         } elseif ($c > 155 && $c < 196) {
  735.             return 'AF';
  736.         } elseif ($c > 195 && $c < 236) {
  737.             return 'D7';
  738.         } else {
  739.             return 'FF';
  740.         }
  741.     }
  742.     
  743.     /**
  744.      * Generate the ANSI/XTERM format string for the format information passed
  745.      * 
  746.      * @param string|integer$fgColor The foreground color to use for the format string
  747.      * @param string|integer$bgColor The background color to use for the format string
  748.      * @param string $style The style to use for the format string
  749.      * 
  750.      * @return string 
  751.      */
  752.     private function GetFormatString($fgColor, $bgColor, $style) {
  753.         // Initialize the format arrays
  754.         $formats = array(
  755.             'foreground'    =>    '',
  756.             'background'    =>    '',
  757.             'style'    =>    ''
  758.         );
  759.         $xFormats = array(
  760.             'foreground'    =>    '',
  761.             'background'    =>    ''
  762.         );
  763.         
  764.         // Check the foreground color
  765.         if (array_key_exists($fgColor, $this->foregroundColors)) {
  766.             $formats['foreground'] = $this->foregroundColors[$fgColor];
  767.         } else {
  768.             if ($this->xtermSupport === true) {
  769.                 // Try this as an xterm color
  770.                 $fgColor = strtoupper($fgColor);
  771.                 
  772.                 if (array_key_exists($fgColor, $this->xtermColors)) {
  773.                     // Good xterm color
  774.                     $xFormats['foreground'] = sprintf($this->escape, '38;5;'.$this->xtermColors[$fgColor]);
  775.                 } else {
  776.                     if (strlen($fgColor) == 6 || strlen($fgColor) == 7) {
  777.                         // Convert RGB string to the closest xterm capable color
  778.                         $xFormats['foreground'] = sprintf($this->escape, '38;5;'.$this->ClosestXtermColor($fgColor, true));
  779.                     } elseif (is_int($fgColor) && $fgColor >= 0 && $fgColor <= 255) {
  780.                         // Already passed as xterm color index
  781.                         $xFormats['foreground'] = sprintf($this->escape, '38;5;'.$fgColor);
  782.                     } else {
  783.                         // No valid xterm color found
  784.                         unset($formats['foreground']);
  785.                     }
  786.                 }
  787.             } else {
  788.                 // No valid ANSI or xterm color found
  789.                 unset($formats['foreground']);
  790.             }
  791.         }
  792.         
  793.         // Check the background color
  794.         if (array_key_exists($bgColor, $this->backgroundColors)) {
  795.             $formats['background'] = $this->backgroundColors[$bgColor];
  796.         } else {
  797.             if ($this->xtermSupport === true) {
  798.                 // Try this as an xterm color
  799.                 $bgColor = strtoupper($bgColor);
  800.                 
  801.                 if (array_key_exists($bgColor, $this->xtermColors)) {
  802.                     // Good xterm color
  803.                     $xFormats['background'] = sprintf($this->escape, '38;5;'.$this->xtermColors[$bgColor]);
  804.                 } else {
  805.                     if (is_string($bgColor) && (strlen($bgColor == 6) || strlen($bgColor) == 7)) {
  806.                         // Convert RGB string to the closest xterm capable color
  807.                         $xFormats['background'] = sprintf($this->escape, '48;5;'.$this->ClosestXtermColor($bgColor, false));
